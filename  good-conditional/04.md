function availableColorsByFruit(fruitName) {
    const colors = []
    if (fruitName === 'banana' || fruitName === 'tomato') {
        colors.push('yellow');
    }

    if (fruitName === 'apple' || fruitName === 'tomato' || fruitName === 'grape') {
        colors.push('red');
    }

    if (fruitName === 'grape') {
        colors.push('red');
    }

    colors.push('green');
    return colors;
}
